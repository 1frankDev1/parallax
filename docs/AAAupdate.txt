-- Creación de la tabla audio_items si no existe
CREATE TABLE IF NOT EXISTS audio_items (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    created_at TIMESTAMPTZ DEFAULT now(),
    title TEXT NOT NULL,
    description TEXT,
    category TEXT,
    filename TEXT,
    file_url TEXT NOT NULL
);
