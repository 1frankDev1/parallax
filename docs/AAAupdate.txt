-- SQL para la creación de la tabla de audios (audio_items)
CREATE TABLE IF NOT EXISTS public.audio_items (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    title TEXT NOT NULL,
    description TEXT,
    category TEXT DEFAULT 'General',
    filename TEXT,
    file_url TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT now()
);

-- Habilitar RLS (Row Level Security)
ALTER TABLE public.audio_items ENABLE ROW LEVEL SECURITY;

-- Políticas para la tabla (Permitir lectura y cambios a todos por simplicidad en este entorno)
-- Nota: En producción, se recomienda restringir el acceso según el rol del usuario.
CREATE POLICY "Permitir lectura pública" ON public.audio_items FOR SELECT USING (true);
CREATE POLICY "Permitir inserción pública" ON public.audio_items FOR INSERT WITH CHECK (true);
CREATE POLICY "Permitir actualización pública" ON public.audio_items FOR UPDATE USING (true);
CREATE POLICY "Permitir eliminación pública" ON public.audio_items FOR DELETE USING (true);

-- Notificar a PostgREST para refrescar el esquema
NOTIFY pgrst, 'reload schema';
