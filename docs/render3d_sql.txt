-- SQL para la creación de la tabla de texturas para Render 3D
CREATE TABLE IF NOT EXISTS public.render3d_textures (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    gltf_key TEXT NOT NULL, -- 'flyer', 'poster', 'shirt', 'sticker', 'tableqr', 'xbanner', 'flag', 'banner'
    texture_url TEXT NOT NULL,
    label TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Habilitar RLS (Row Level Security)
ALTER TABLE public.render3d_textures ENABLE ROW LEVEL SECURITY;

-- Políticas para la tabla (Permitir lectura a todos, cambios a todos por simplicidad en este entorno)
CREATE POLICY "Permitir lectura pública" ON public.render3d_textures FOR SELECT USING (true);
CREATE POLICY "Permitir inserción pública" ON public.render3d_textures FOR INSERT WITH CHECK (true);
CREATE POLICY "Permitir eliminación pública" ON public.render3d_textures FOR DELETE USING (true);
CREATE POLICY "Permitir actualización pública" ON public.render3d_textures FOR UPDATE USING (true);
