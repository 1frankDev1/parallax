-- Tabla para almacenar las texturas personalizadas vinculadas a los modelos GLTF
CREATE TABLE render3d_textures (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    gltf_key TEXT NOT NULL,         -- Clave del modelo (p. ej. 'flyer', 'poster', 'sticker', etc.)
    label TEXT NOT NULL,            -- Nombre descriptivo del diseño
    texture_url TEXT NOT NULL,      -- URL de la imagen en Cloudinary
    created_at TIMESTAMPTZ DEFAULT now()
);
